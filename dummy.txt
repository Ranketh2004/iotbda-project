commit_with_date () {
  DATE=$1
  COUNT=$2
  for ((i=0;i<$COUNT;i++))
  do
    echo "// auth update $RANDOM" >> dummy.txt

    HOUR=$((10 + RANDOM % 10))
    MIN=$((RANDOM % 60))

    MSG=${messages[$RANDOM % ${#messages[@]}]}

    GIT_AUTHOR_DATE="$DATE $HOUR:$MIN:00" \
    GIT_COMMITTER_DATE="$DATE $HOUR:$MIN:00" \
    git add . && git commit -m "$MSG"
  done
}
